98 def str_to_display(data, desc=None):
99 # type: (Union[bytes, Text], Optional[str]) -> Text
100 """
101 For display or logging purposes, convert a bytes object (or text) to
102 text (e.g. unicode in Python 2) safe for output.
103  
104 :param desc: An optional phrase describing the input data, for use in
105 the log message if a warning is logged. Defaults to "Bytes object".
106  
107 This function should never error out and so can take a best effort
108 approach. It is okay to be lossy if needed since the return value is
109 just for display.
110  
111 We assume the data is in the locale preferred encoding. If it won't
112 decode properly, we warn the user but decode as best we can.
113  
114 We also ensure that the output can be safely written to standard output
115 without encoding errors.
116 """
117 if isinstance(data, text_type):
118 return data
119  
120 # Otherwise, data is a bytes object (str in Python 2).
121 # First, get the encoding we assume. This is the preferred
122 # encoding for the locale, unless that is not found, or
123 # it is ASCII, in which case assume UTF-8
124 encoding = locale.getpreferredencoding()
125 if (not encoding) or codecs.lookup(encoding).name == "ascii":
126 encoding = "utf-8"
127  
128 # Now try to decode the data - if we fail, warn the user and
129 # decode with replacement.
130 try:
131 decoded_data = data.decode(encoding)
132 except UnicodeDecodeError:
133 logger.warning(
134 '%s does not appear to be encoded as %s',
135 desc or 'Bytes object',
136 encoding,
137 )
138 decoded_data = data.decode(encoding, errors=backslashreplace_decode)
139  
140 # Make sure we can print the output, by encoding it to the output
141 # encoding with replacement of unencodable characters, and then
142 # decoding again.
143 # We use stderr's encoding because it's less likely to be
144 # redirected and if we don't find an encoding we skip this
145 # step (on the assumption that output is wrapped by something
146 # that won't fail).
147 # The double getattr is to deal with the possibility that we're
148 # being called in a situation where sys.__stderr__ doesn't exist,
149 # or doesn't have an encoding attribute. Neither of these cases
150 # should occur in normal pip use, but there's no harm in checking
151 # in case people use pip in (unsupported) unusual situations.
152 output_encoding = getattr(getattr(sys, "__stderr__", None),
153 "encoding", None)
154  
155 if output_encoding:
156 output_encoded = decoded_data.encode(
157 output_encoding,
158 errors="backslashreplace"
159 )
160 decoded_data = output_encoded.decode(output_encoding)
161  
162 return decoded_data

172 def get_path_uid(path):
173 # type: (str) -> int
174 """
175 Return path's uid.
176  
177 Does not follow symlinks:
178 https://github.com/pypa/pip/pull/935#discussion_r5307003
179  
180 Placed this function in compat due to differences on AIX and
181 Jython, that should eventually go away.
182  
183 :raises OSError: When path is a symlink or can't be read.
184 """
185 if hasattr(os, 'O_NOFOLLOW'):
186 fd = os.open(path, os.O_RDONLY | os.O_NOFOLLOW)
187 file_uid = os.fstat(fd).st_uid
188 os.close(fd)
189 else: # AIX and Jython
190 # WARNING: time of check vulnerability, but best we can do w/o NOFOLLOW
191 if not os.path.islink(path):
192 # older versions of Jython don't have `os.fstat`
193 file_uid = os.stat(path).st_uid
194 else:
195 # raise OSError for parity with os.O_NOFOLLOW above
196 raise OSError(
197 "{} is a symlink; Will not return uid for symlinks".format(
198 path)
199 )
200 return file_uid

205 """
206 Expand ~ and ~user constructions.

208 Includes a workaround for https://bugs.python.org/issue14768
209 """
210 expanded = os.path.expanduser(path)

212 expanded = expanded[1:]
