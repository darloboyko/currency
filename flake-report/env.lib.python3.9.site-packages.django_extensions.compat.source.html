14 def load_tag_library(libname):
15 """
16 Load a templatetag library on multiple Django versions.
17  
18 Returns None if the library isn't loaded.
19 """
20 from django.template.backends.django import get_installed_libraries
21 from django.template.library import InvalidTemplateLibrary
22 try:
23 lib = get_installed_libraries()[libname]
24 lib = importlib.import_module(lib).register
25 return lib
26 except (InvalidTemplateLibrary, KeyError):
27 return None

40 class UnicodeWriter:
41 """
42 CSV writer which will write rows to CSV file "f",
43 which is encoded in the given encoding.
44 We are using this custom UnicodeWriter for python versions 2.x
45 """

47 def __init__(self, f, dialect=csv.excel, encoding="utf-8", **kwds):
48 self.queue = BytesIO()
49 self.writer = csv.writer(self.queue, dialect=dialect, **kwds)
50 self.stream = f
51 self.encoder = codecs.getincrementalencoder(encoding)()
52  
53 def writerow(self, row):
54 self.writer.writerow([s.encode("utf-8") for s in row])
55 # Fetch UTF-8 output from the queue ...
56 data = self.queue.getvalue()
57 data = data.decode("utf-8")
58 # ... and reencode it into the target encoding
59 data = self.encoder.encode(data)
60 # write to the target stream
61 self.stream.write(data)
62 # empty queue
63 self.queue.truncate(0)
