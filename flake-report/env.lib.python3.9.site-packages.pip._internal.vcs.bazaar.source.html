94 @classmethod
95 def get_remote_url(cls, location):
96 urls = cls.run_command(
97 ['info'], show_stdout=False, stdout_only=True, cwd=location
98 )
99 for line in urls.splitlines():
100 line = line.strip()
101 for x in ('checkout of branch: ',
102 'parent branch: '):
103 if line.startswith(x):
104 repo = line.split(x)[1]
105 if cls._is_local_repository(repo):
106 return path_to_url(repo)
107 return repo
108 return None
