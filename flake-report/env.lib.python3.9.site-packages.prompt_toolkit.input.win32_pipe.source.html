16 class Win32PipeInput(_Win32InputBase, PipeInput):
17 """
18 This is an input pipe that works on Windows.
19 Text or bytes can be feed into the pipe, and key strokes can be read from
20 the pipe. This is useful if we want to send the input programmatically into
21 the application. Mostly useful for unit testing.
22  
23 Notice that even though it's Windows, we use vt100 escape sequences over
24 the pipe.
25  
26 Usage::
27  
28 input = Win32PipeInput()
29 input.send_text('inputdata')
30 """
31  
32 _id = 0

34 def __init__(self) -> None:
35 super().__init__()
36 # Event (handle) for registering this input in the event loop.
37 # This event is set when there is data available to read from the pipe.
38 # Note: We use this approach instead of using a regular pipe, like
39 # returned from `os.pipe()`, because making such a regular pipe
40 # non-blocking is tricky and this works really well.
41 self._event = create_win32_event()
42  
43 self._closed = False
44  
45 # Parser for incoming keys.
46 self._buffer: List[KeyPress] = [] # Buffer to collect the Key objects.
47 self.vt100_parser = Vt100Parser(lambda key: self._buffer.append(key))
48  
49 # Identifier for every PipeInput for the hash.
50 self.__class__._id += 1
51 self._id = self.__class__._id

96 Flush pending keys and return them.
