136 @classmethod
137 def parse_options(cls, options):
138 """Parse option values from Flake8's OptionManager."""
139 if options.builtins:
140 cls.builtIns = cls.builtIns.union(options.builtins)
141 cls.with_doctest = options.doctests
142  
143 included_files = []
144 for included_file in options.include_in_doctest:
145 if included_file == "":
146 continue
147 if not included_file.startswith((os.sep, "./", "~/")):
148 included_files.append(f"./{included_file}")
149 else:
150 included_files.append(included_file)
151 cls.include_in_doctest = utils.normalize_paths(included_files)
152  
153 excluded_files = []
154 for excluded_file in options.exclude_from_doctest:
155 if excluded_file == "":
156 continue
157 if not excluded_file.startswith((os.sep, "./", "~/")):
158 excluded_files.append(f"./{excluded_file}")
159 else:
160 excluded_files.append(excluded_file)
161 cls.exclude_from_doctest = utils.normalize_paths(excluded_files)
162  
163 inc_exc = set(cls.include_in_doctest).intersection(
164 cls.exclude_from_doctest
